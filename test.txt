class Calculator {
    public:
        decl factor as int;
        decl memory as float;

        Calculator(decl f as int) {
            factor = f;
            memory = 0.0;
        }

        decl addValues(decl a as int, decl b as int) as int {
            decl result as int;
            result = (a + b) * factor;
            return result;
        }
};

decl myCalc as Calculator;

decl checkParity(decl val as int) as bool {
    decl isEven as bool;
    if ((val % 2) == 0) {
        isEven = true;
    } else {
        isEven = false;
    }
    return isEven;
}
